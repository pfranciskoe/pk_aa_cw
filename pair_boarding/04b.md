function sillyYears(year){
    let sillys = [];
    let down = year;
    let up = year + 1;
    while(sillys.length < 10){
        if (down > 999 &&isSilly(down)){
            sillys.push(down);

        if (up < 9999 && isSilly(up)){
            sillys.push(up);
        }
        down--;
        up++;
    }
    return sillys.slice(0,10);
}

    let first = year % 100;
    let last = Math.floor(year/100);
    let middle = Math.floor((year % 1000) / 10);
