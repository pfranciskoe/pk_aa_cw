    funciton digitalRoot(num){
        let result = num
        while (result >= 10) {
            result = addDigits(result);
        }
        return result
    }

    function addDigits(num){
        let oldNumber = num;
        let newNumber = 0;
        while (oldNumber >= 10) {
            newNumber += oldNumber % 10;
            oldNumber = Math.floor(oldNumber / 10);
        }
        newNumber += oldNumber;
        return newNumber
    }
